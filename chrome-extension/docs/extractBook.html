	extractEntryByName: function(name, callback) {
		var found = false;
		for (var i=0; i < this.zip.entries.length; i++) {
			if(this.zip.entries[i].name === name) {
				found = true;
				this.zip.entries[i].extract(callback);
				break;
			}
		}
		if(!found) {
			throw ("asked to extract non-existent zip-entry: " + name);
		}
	},
