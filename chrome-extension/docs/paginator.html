	findPrerenderStart: function() {
		var i = 0;

			i -= 1;

		return i + 1; // sloppy fix for an off by one error

	shouldRenderAsFixed: function(spineItem) {

	shouldPreRender: function(sec) {
		return sec && this.shouldRenderAsFixed(sec);
