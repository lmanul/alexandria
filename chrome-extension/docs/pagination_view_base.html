    injectMathJax: function (iframe) {
		var doc = iframe.contentDocument;

		script.type = "text/javascript";
		script.src = MathJax.Hub.config.root+"/MathJax.js?config=readium-iframe";

    injectLinkHandler: function(iframe) {
